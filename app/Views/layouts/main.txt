</head>
<body class="app-dark">
    <?= $this->include('partials/navbar') ?>

    <main class="app-main py-3">
        <div class="container-fluid app-container px-3 px-xl-4">
            <div id="ajaxFlashHost" class="app-flash-host">
                <?= $this->include('partials/flash') ?>
            </div>
            <?= $this->renderSection('content') ?>
        </div>
    </main>

    <script>
        window.BGApp = {
            csrfTokenName: '<?= esc(csrf_token()) ?>',
            csrfHash: '<?= esc(csrf_hash()) ?>',
            baseUrl: '<?= site_url('/') ?>',

            ensureFormCsrf(form) {
                if (!form || !this.csrfTokenName || !this.csrfHash) {
                    return;
                }

                let input = form.querySelector(`input[name="${this.csrfTokenName}"]`);

                if (!input) {
                    form.querySelectorAll('input[type="hidden"]').forEach((hidden) => {
                        if (/csrf/i.test(hidden.name || '')) {
                            hidden.remove();
                        }
                    });

                    input = document.createElement('input');
                    input.type = 'hidden';
                    input.name = this.csrfTokenName;
                    form.prepend(input);
                }

                input.value = this.csrfHash;
            },

            syncAllPostForms() {
                document.querySelectorAll('form').forEach((form) => {
                    const method = (form.getAttribute('method') || 'get').toLowerCase();
                    if (method === 'post') {
                        this.ensureFormCsrf(form);
                    }
                });
            },

            showFlash(message, type = 'success') {
                const host = document.getElementById('ajaxFlashHost');
                if (!host || !message) return;

                const classMap = {
                    success: 'alert-success',
                    error: 'alert-danger',
                    warning: 'alert-warning',
                    info: 'alert-info'
                };

                const flash = document.createElement('div');
                flash.className = `alert ${classMap[type] || 'alert-secondary'} alert-dismissible fade show ajax-flash`;
                flash.setAttribute('role', 'alert');
                flash.innerHTML = '<button type="button" class="btn-close" data-bs-dismiss="alert" aria-label="Close"></button>';
                flash.prepend(document.createTextNode(message));
                host.appendChild(flash);

                window.BGApp.armFlash(flash);
            },

            armFlash(flash) {
                if (!flash) return;

                window.setTimeout(() => {
                    flash.classList.add('is-hiding');
                }, 3200);

                window.setTimeout(() => {
                    if (window.bootstrap?.Alert) {
                        window.bootstrap.Alert.getOrCreateInstance(flash).close();
                        return;
                    }

                    flash.remove();
                }, 3800);
            },

            updateCsrf(tokenName, tokenHash) {
                if (!tokenName || !tokenHash) return;

                this.csrfTokenName = tokenName;
                this.csrfHash = tokenHash;
                this.syncAllPostForms();
            }
        };

        document.addEventListener('DOMContentLoaded', () => {
            window.BGApp.syncAllPostForms();

            document.querySelectorAll('#ajaxFlashHost .alert').forEach((flash) => {
                window.BGApp.armFlash(flash);
            });
        });

        document.addEventListener('submit', (event) => {
            const form = event.target;
            if (!(form instanceof HTMLFormElement)) {
                return;
            }

            const method = (form.getAttribute('method') || 'get').toLowerCase();
            if (method === 'post') {
                window.BGApp.ensureFormCsrf(form);
            }
        }, true);
    </script>
